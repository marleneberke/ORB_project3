272 # If A is grouped, so that each unique element is in a contiguous group, then we only
273 # need to keep track of one element at a time. We replace the elements of A with the
274 # unique elements that we see in the order that we see them. Once we have iterated
275 # through A, we resize A based on the number of unique elements that we see.
276 function _groupedunique!(A::AbstractVector)
277 isempty(A) && return A
278 idxs = eachindex(A)
279 y = first(A)
280 # We always keep the first element
281 it = iterate(idxs, iterate(idxs)[2])
282 count = 1
283 for x in Iterators.drop(A, 1)
284 if !isequal(x, y)
285 y = A[it[1]] = x
286 count += 1
287 it = iterate(idxs, it[2])
288 end
289 end

291 end

590 function _replace!(new::Callable, res::T, A::T,
591 count::Int) where T<:Union{AbstractDict,AbstractSet}
592 c = 0
593 if res === A # cannot replace elements while iterating over A
594 repl = Pair{eltype(A),eltype(A)}[]
595 for x in A
596 y = new(x)
597 if x !== y
598 push!(repl, x => y)
599 c += 1
600 end
601 c == count && break
602 end
603 for oldnew in repl
604 pop!(res, askey(first(oldnew), res))
605 end
606 for oldnew in repl
607 push!(res, last(oldnew))
608 end
609 else
610 for x in A
611 y = new(x)
612 if x !== y
613 pop!(res, askey(x, res))
614 push!(res, y)
615 c += 1
616 end
617 c == count && break
618 end
619 end
