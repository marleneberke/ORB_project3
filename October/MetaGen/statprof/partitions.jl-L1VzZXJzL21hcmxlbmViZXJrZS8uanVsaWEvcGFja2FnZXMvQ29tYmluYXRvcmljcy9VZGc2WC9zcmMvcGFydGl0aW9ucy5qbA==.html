394 """
395 prevprod(a::Vector{Int}, x)
396
397 Previous integer not greater than `x` that can be written as ``\\prod k_i^{p_i}`` for
398 integers ``p_1``, ``p_2``, etc.
399
400 For integers ``i_1``, ``i_2``, ``i_3``, this is equivalent to finding the largest ``x``
401 such that
402 ```math
403 i_1^n_1 i_2^n_2 i_3^n_3 \\leq x
404 ```
405 for integers ``n_1``, ``n_2``, ``n_3``.
406 """
407 function prevprod(a::Vector{Int}, x)
408 if x > typemax(Int)
409 throw(ArgumentError("unsafe for x > typemax(Int), got $x"))
410 end
411 k = length(a)
412 v = ones(Int, k) # current value of each counter
413 mx = [nextpow(ai,x) for ai in a] # allow each counter to exceed p (sentinel)
414 first = Int(prevpow(a[1], x)) # start at best case in first factor
415 v[1] = first
416 p::widen(Int) = first
417 best = p
418 icarry = 1
419
420 while v[end] < mx[end]
421 while p <= x
422 best = p > best ? p : best
423 p *= a[1]
424 v[1] *= a[1]
425 end
426 if p > x
427 carrytest = true
428 while carrytest
429 p = div(p, v[icarry])
430 v[icarry] = 1
431 icarry += 1
432 p *= a[icarry]
433 v[icarry] *= a[icarry]
434 carrytest = v[icarry] > mx[icarry] && icarry < k
435 end
436 if p <= x
437 icarry = 1
438 end
439 end
440 end
441 best = x >= p > best ? p : best
442 return Int(best)
443 end

489 function _ncpart!(a::Int, b::Int, nn::Int, x::Vector, partitions::Vector)
490 n = b - a + 1
491 for k = 1:n, root in CoolLexCombinations(n, k)
492 root .+= a - 1
493 #Abort if construction is out of lex order
494 !isempty(x) && _cmp(x[end], root) == 1 && return
495
496 #Save if we've filled all the holes
497 sofar = Vector{Int}[x..., root]
498 ssofaru = sort!(union(sofar...))
499 if length(ssofaru) == nn && ssofaru == collect(1:nn)
500 push!(partitions, sofar)
501 return
502 end
503
504 #otherwise patch all remaining holes
505 blob = [ssofaru; nn + 1]
506 for l = 1:length(blob)-1
507 ap, bp = blob[l] + 1, blob[l+1] - 1
508 ap <= bp && _ncpart!(ap, bp, nn, sofar, partitions)
