320 function repeat(c::Char, r::Integer)
321 r == 0 && return ""
322 r < 0 && throw(ArgumentError("can't repeat a character $r times"))
323 u = bswap(reinterpret(UInt32, c))
324 n = 4 - (leading_zeros(u | 0xff) >> 3)
325 s = _string_n(n*r)
326 p = pointer(s)
327 GC.@preserve s if n == 1
328 ccall(:memset, Ptr{Cvoid}, (Ptr{UInt8}, Cint, Csize_t), p, u % UInt8, r)
329 elseif n == 2
330 p16 = reinterpret(Ptr{UInt16}, p)
331 for i = 1:r
332 unsafe_store!(p16, u % UInt16, i)
333 end
334 elseif n == 3
335 b1 = (u >> 0) % UInt8
336 b2 = (u >> 8) % UInt8
337 b3 = (u >> 16) % UInt8
338 for i = 0:r-1
339 unsafe_store!(p, b1, 3i + 1)
340 unsafe_store!(p, b2, 3i + 2)
341 unsafe_store!(p, b3, 3i + 3)
342 end
343 elseif n == 4
344 p32 = reinterpret(Ptr{UInt32}, p)
345 for i = 1:r
346 unsafe_store!(p32, u, i)
347 end
348 end
349 return s
350 end
