50 function _foldl_impl(op::OP, init, itr) where {OP}
51 # Unroll the while loop once; if init is known, the call to op may
52 # be evaluated at compile time
53 y = iterate(itr)
54 y === nothing && return init
55 v = op(init, y[1])
56 while true
57 y = iterate(itr, y[2])
58 y === nothing && break
59 v = op(v, y[1])
60 end
61 return v

119 """
120 _xfadjoint(op, itr) -> op′, itr′
121
122 Given a pair of reducing function `op` and an iterator `itr`, return a pair
123 `(op′, itr′)` of similar types. If the iterator `itr` is transformed by an
124 iterator transform `ixf` whose adjoint transducer `xf` is known, `op′ = xf(op)`
125 and `itr′ = ixf⁻¹(itr)` is returned. Otherwise, `op` and `itr` are returned
126 as-is. For example, transducer `rf -> MappingRF(f, rf)` is the adjoint of
127 iterator transform `itr -> Generator(f, itr)`.
128
129 Nested iterator transforms are converted recursively. That is to say,
130 given `op` and
131
132 itr = (ixf₁ ∘ ixf₂ ∘ ... ∘ ixfₙ)(itr′)
133
134 what is returned is `itr′` and

223 # `mapreduce_impl()` is called by `mapreduce()` (via `_mapreduce()`, when `A`
224 # supports linear indexing) and does actual calculations (for `A[ifirst:ilast]` subset).
225 # For efficiency, no parameter validity checks are done, it's the caller's responsibility.
226 # `ifirst:ilast` range is assumed to be a valid non-empty subset of `A` indices.
227
228 # This is a generic implementation of `mapreduce_impl()`,
229 # certain `op` (e.g. `min` and `max`) may have their own specialized versions.
230 @noinline function mapreduce_impl(f, op, A::AbstractArray, ifirst::Integer, ilast::Integer, blksize::Int)
231 if ifirst == ilast
232 @inbounds a1 = A[ifirst]
233 return mapreduce_first(f, op, a1)
234 elseif ifirst + blksize > ilast
235 # sequential portion
236 @inbounds a1 = A[ifirst]
237 @inbounds a2 = A[ifirst+1]
238 v = op(f(a1), f(a2))
239 @simd for i = ifirst + 2 : ilast
240 @inbounds ai = A[i]
241 v = op(v, f(ai))
242 end
243 return v
244 else
245 # pairwise portion
246 imid = (ifirst + ilast) >> 1
247 v1 = mapreduce_impl(f, op, A, ifirst, imid, blksize)
248 v2 = mapreduce_impl(f, op, A, imid+1, ilast, blksize)
249 return op(v1, v2)
250 end
251 end

571 # short circuit in case of NaN

573 v2 == v2 || return v2
574 v3 == v3 || return v3

579 v3 = _fast(op, v3, f(A[i+2]))

662 ## all & any
663
664 """
665 any(itr) -> Bool
666
667 Test whether any elements of a boolean collection are `true`, returning `true` as
668 soon as the first `true` value in `itr` is encountered (short-circuiting).
669
670 If the input contains [`missing`](@ref) values, return `missing` if all non-missing
671 values are `false` (or equivalently, if the input contains no `true` value), following
672 [three-valued logic](https://en.wikipedia.org/wiki/Three-valued_logic).
673
674 # Examples
675 ```jldoctest
676 julia> a = [true,false,false,true]
677 4-element Array{Bool,1}:
678 1
679 0
680 0
681 1
682
683 julia> any(a)
684 true
685
686 julia> any((println(i); v) for (i, v) in enumerate(a))
687 1
688 true
689
690 julia> any([missing, true])
691 true
692
693 julia> any([false, missing])
694 missing
695 ```
696 """
697 any(itr) = any(identity, itr)
698
699 """
700 all(itr) -> Bool
701
702 Test whether all elements of a boolean collection are `true`, returning `false` as
703 soon as the first `false` value in `itr` is encountered (short-circuiting).
704
705 If the input contains [`missing`](@ref) values, return `missing` if all non-missing
706 values are `true` (or equivalently, if the input contains no `false` value), following
707 [three-valued logic](https://en.wikipedia.org/wiki/Three-valued_logic).
708
709 # Examples
710 ```jldoctest
711 julia> a = [true,false,false,true]
712 4-element Array{Bool,1}:
713 1
714 0
715 0
716 1
717
718 julia> all(a)
719 false
720
721 julia> all((println(i); v) for (i, v) in enumerate(a))
722 1
723 2
724 false
725
726 julia> all([missing, false])
727 false
728
729 julia> all([true, missing])
730 missing
731 ```
732 """
733 all(itr) = all(identity, itr)
734
735 """
736 any(p, itr) -> Bool
737
738 Determine whether predicate `p` returns `true` for any elements of `itr`, returning
739 `true` as soon as the first item in `itr` for which `p` returns `true` is encountered
740 (short-circuiting).
741
742 If the input contains [`missing`](@ref) values, return `missing` if all non-missing
743 values are `false` (or equivalently, if the input contains no `true` value), following
744 [three-valued logic](https://en.wikipedia.org/wiki/Three-valued_logic).
745
746 # Examples
747 ```jldoctest
748 julia> any(i->(4<=i<=6), [3,5,7])
749 true
750
751 julia> any(i -> (println(i); i > 3), 1:10)
752 1
753 2
754 3
755 4
756 true
757
758 julia> any(i -> i > 0, [1, missing])
759 true
760
761 julia> any(i -> i > 0, [-1, missing])
762 missing
763
764 julia> any(i -> i > 0, [-1, 0])
765 false
766 ```
767 """
768 any(f, itr) = _any(f, itr, :)
769
770 function _any(f, itr, ::Colon)
771 anymissing = false
772 for x in itr
773 v = f(x)
774 if ismissing(v)
775 anymissing = true
776 elseif v
777 return true
778 end
779 end
780 return anymissing ? missing : false
781 end

783 """
784 all(p, itr) -> Bool
785
786 Determine whether predicate `p` returns `true` for all elements of `itr`, returning
787 `false` as soon as the first item in `itr` for which `p` returns `false` is encountered
788 (short-circuiting).
789
790 If the input contains [`missing`](@ref) values, return `missing` if all non-missing
791 values are `true` (or equivalently, if the input contains no `false` value), following
792 [three-valued logic](https://en.wikipedia.org/wiki/Three-valued_logic).
793
794 # Examples
795 ```jldoctest
796 julia> all(i->(4<=i<=6), [4,5,6])
797 true
798
799 julia> all(i -> (println(i); i < 3), 1:10)
800 1
801 2
802 3
803 false
804
805 julia> all(i -> i > 0, [1, missing])
806 missing
807
808 julia> all(i -> i > 0, [-1, missing])
809 false
810
811 julia> all(i -> i > 0, [1, 2])
812 true
813 ```
814 """
815 all(f, itr) = _all(f, itr, :)
816
817 function _all(f, itr, ::Colon)
818 anymissing = false
819 for x in itr
820 v = f(x)
821 if ismissing(v)
822 anymissing = true
823 # this syntax allows throwing a TypeError for non-Bool, for consistency with any
824 elseif v
825 continue
826 else
827 return false
828 end
829 end
830 return anymissing ? missing : true
831 end

833 ## count
834
835 """
836 count(p, itr) -> Integer
837 count(itr) -> Integer
838
839 Count the number of elements in `itr` for which predicate `p` returns `true`.
840 If `p` is omitted, counts the number of `true` elements in `itr` (which
841 should be a collection of boolean values).
842
843 # Examples
844 ```jldoctest
845 julia> count(i->(4<=i<=6), [2,3,4,5,6])
846 3
847
848 julia> count([true, false, true, true])
849 3
850 ```
851 """
852 function count(pred, itr)
853 n = 0
854 for x in itr
855 n += pred(x)::Bool
856 end
857 return n
858 end
859 function count(pred, a::AbstractArray)
860 n = 0
861 for i in eachindex(a)
862 @inbounds n += pred(a[i])::Bool
863 end
864 return n
865 end
866 count(itr) = count(identity, itr)
