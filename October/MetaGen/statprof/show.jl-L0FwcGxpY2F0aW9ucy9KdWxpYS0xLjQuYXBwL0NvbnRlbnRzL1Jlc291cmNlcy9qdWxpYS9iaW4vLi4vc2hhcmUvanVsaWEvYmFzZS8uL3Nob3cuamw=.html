771 ## Abstract Syntax Tree (AST) printing ##
772
773 # Summary:
774 # print(io, ex) defers to show_unquoted(io, ex)
775 # show(io, ex) defers to show_unquoted(io, QuoteNode(ex))
776 # show_unquoted(io, ex) does the heavy lifting
777 #
778 # AST printing should follow two rules:
779 # 1. Meta.parse(string(ex)) == ex
780 # 2. eval(Meta.parse(repr(ex))) == ex
781 #
782 # Rule 1 means that printing an expression should generate Julia code which
783 # could be reparsed to obtain the original expression. This code should be
784 # unambiguous and as readable as possible.
785 #
786 # Rule 2 means that showing an expression should generate a quoted version of
787 # print’s output. Parsing and then evaling this output should return the
788 # original expression.
789 #
790 # This is consistent with many other show methods, i.e.:
791 # show(Set([1,2,3])) # ==> "Set{Int64}([2,3,1])"
792 # eval(Meta.parse("Set{Int64}([2,3,1])”) # ==> An actual set
793 # While this isn’t true of ALL show methods, it is of all ASTs.
794
795 const ExprNode = Union{Expr, QuoteNode, Slot, LineNumberNode, SSAValue,
796 GotoNode, GlobalRef, PhiNode, PhiCNode, UpsilonNode,
797 Core.Compiler.GotoIfNot, Core.Compiler.ReturnNode}
798 # Operators have precedence levels from 1-N, and show_unquoted defaults to a
799 # precedence level of 0 (the fourth argument). The top-level print and show
800 # methods use a precedence of -1 to specially allow space-separated macro syntax.
801 # IOContext(io, :unquote_fallback => false) tells show_unquoted to treat any
802 # Expr whose head is :$ as if it is inside a quote, preventing fallback to the
803 # "unhandled" case: this is used by print/string to be lawful to Rule 1 above.
804 # On the countrary, show/repr have to follow Rule 2, requiring any Expr whose
805 # head is :$ and which is not inside a quote to fallback to the "unhandled" case:
806 # this is behavior is triggered by IOContext(io, :unquote_fallback => true)
807 print( io::IO, ex::ExprNode) = (show_unquoted(IOContext(io, :unquote_fallback => false), ex, 0, -1); nothing)
808 show( io::IO, ex::ExprNode) = show_unquoted_quote_expr(IOContext(io, :unquote_fallback => true), ex, 0, -1, 0)
809 show_unquoted(io::IO, ex) = show_unquoted(io, ex, 0, 0)
810 show_unquoted(io::IO, ex, indent::Int) = show_unquoted(io, ex, indent, 0)
811 show_unquoted(io::IO, ex, ::Int,::Int) = show(io, ex)
812 show_unquoted(io::IO, ex, indent::Int, prec::Int, ::Int) = show_unquoted(io, ex, indent, prec)

885 """
886 operator_precedence(s::Symbol)
887
888 Return an integer representing the precedence of operator `s`, relative to
889 other operators. Higher-numbered operators take precedence over lower-numbered
890 operators. Return `0` if `s` is not a valid operator.
891
892 # Examples
893 ```jldoctest
894 julia> Base.operator_precedence(:+), Base.operator_precedence(:*), Base.operator_precedence(:.)

896
897 julia> Base.operator_precedence(:sin), Base.operator_precedence(:+=), Base.operator_precedence(:(=)) # (Note the necessary parens on `:(=)`)
898 (0, 1, 1)
899 ```
900 """
901 operator_precedence(s::Symbol) = Int(ccall(:jl_operator_precedence, Cint, (Cstring,), s))
902 operator_precedence(x::Any) = 0 # fallback for generic expression nodes
903 const prec_assignment = operator_precedence(:(=))
904 const prec_pair = operator_precedence(:(=>))
905 const prec_control_flow = operator_precedence(:(&&))
906 const prec_arrow = operator_precedence(:(-->))
907 const prec_comparison = operator_precedence(:(>))
908 const prec_power = operator_precedence(:(^))
909 const prec_decl = operator_precedence(:(::))

911 """
912 operator_associativity(s::Symbol)
913
914 Return a symbol representing the associativity of operator `s`. Left- and right-associative
915 operators return `:left` and `:right`, respectively. Return `:none` if `s` is non-associative
916 or an invalid operator.
917
918 # Examples
919 ```jldoctest
920 julia> Base.operator_associativity(:-), Base.operator_associativity(:+), Base.operator_associativity(:^)
921 (:left, :none, :right)
922
923 julia> Base.operator_associativity(:⊗), Base.operator_associativity(:sin), Base.operator_associativity(:→)
924 (:left, :none, :right)
925 ```
926 """
927 function operator_associativity(s::Symbol)
928 if operator_precedence(s) in (prec_arrow, prec_assignment, prec_control_flow, prec_pair, prec_power) ||
929 (isunaryoperator(s) && !is_unary_and_binary_operator(s)) || s === :<|
930 return :right
931 elseif operator_precedence(s) in (0, prec_comparison) || s in (:+, :++, :*)
932 return :none
933 end
934 return :left
935 end
